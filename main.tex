% =============================
%   Авто-нумерация задач + ссылки
% =============================
\newcounter{task}
\renewcommand{\thetask}{\arabic{task}}

% --- Независимые поднумерации внутри каждой задачи ---
\newcounter{subtasknum}[task] % (1), (2), ... сброс при каждом task
\renewcommand{\thesubtasknum}{(\arabic{subtasknum})}

\newcounter{subtaskabc}[task] % а), б), ... сброс при каждом task
\renewcommand{\thesubtaskabc}{\asbuk{subtaskabc})}

% \subtasknum -> (1) (2) ...
% \subtasknum[Название] -> (1) Название
\newcommand{\subtasknum}[1][]{%
  \refstepcounter{subtasknum}%
  \par\noindent\textbf{\thesubtasknum}%
  \if\relax\detokenize{#1}\relax\else\ \textbf{#1}\fi%
  \par
}

% \subtaskabc -> а) б) ...
% \subtaskabc[Название] -> а) Название
\newcommand{\subtaskabc}[1][]{%
  \refstepcounter{subtaskabc}%
  \par\noindent\textbf{\thesubtaskabc}%
  \if\relax\detokenize{#1}\relax\else\ \textbf{#1}\fi%
  \par
}

% Использование:
% \task            -> "Задача 1."
% \task[Название]  -> "Задача 2. Название"
% Можно ставить \label{t:...} сразу после \task и ссылаться \ref{t:...}
\newcommand{\task}[1][]{%
  \refstepcounter{task}%
  % На всякий случай сбрасываем явно (хотя [task] уже делает это автоматически)
  \setcounter{subtasknum}{0}%
  \setcounter{subtaskabc}{0}%
  \vspace{1em}%
  \noindent\textbf{Задача \thetask.}%
  \if\relax\detokenize{#1}\relax\else\ \textbf{#1}\fi%
  \par
}
