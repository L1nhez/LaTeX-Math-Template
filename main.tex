% -------- Расширенная матрица (A | b) --------
% Использование: \aug{1 & 2 \\ 3 & 4}{5 \\ 6}
% Расширенная матрица через |
% Использование: \augmat{ A }{ b }
% где A — левая часть, b — правая колонка
\newcommand{\augmat}[2]{%
\left[
\begin{array}{c|c}
#1 & #2
\end{array}
\right]
}
% Важно: команда \aug рассчитана на 2 столбца в A (cc|c).
% Ниже есть универсальный вариант \Aug.

% Универсальная расширенная матрица:
% \Aug{<cols of A>}{<A entries>}{<cols of b>}{<b entries>}
% Пример: \Aug{ccc}{1&0&2\\0&1&-1}{c}{3\\4}
\newcommand{\Aug}[4]{%
\left[\begin{array}{#1|#3}
#2 & #4
\end{array}\right]
}
